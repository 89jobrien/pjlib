    /var/lib/minibox/
    │
    ├── containers/
    │   └── <container-id>/
    │       ├── pid                    # Process ID file
    │       ├── overlay/
    │       │   ├── lower/            # Image layers (read-only)
    │       │   ├── upper/            # Container changes (writable)
    │       │   ├── work/             # Overlay working directory
    │       │   └── merged/           # Combined view → container root
    │       └── rootfs/               # Symlink to merged/
    │
    └── images/
        └── <image-name>/
            ├── manifest.json
            └── layers/
                ├── <sha256>.tar.gz
                └── ...

    /sys/fs/cgroup/
    └── minibox/
        └── <container-id>/
            ├── cgroup.procs          # PIDs in cgroup
            ├── memory.max            # Memory limit
            ├── memory.current        # Current memory usage
            └── cpu.weight            # CPU scheduling weight

    /var/run/
    └── minibox.sock                  # Daemon Unix socket
